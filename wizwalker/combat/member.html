 52    async def get_health_text_window(self) -> "wizwalker.memory.DynamicWindow":
 53        """
 54        Get the health text window
 55        Useful for targeting
 56        """
 57        possible = await wizwalker.utils.maybe_wait_for_any_value_with_timeout(

 61        if possible:
 62            return possible[0]
