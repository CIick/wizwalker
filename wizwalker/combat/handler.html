 28        """
 29        Handles an entire combat interaction
 30        """
 31        while await self.in_combat():
 32            await self.wait_for_planning_phase()
 33            round_number = await self.round_number()

 35            await self.handle_round()

 87        """
 88        Wait for the round number to change

242        """
243        Get enchants that increse the damage of spells
244
245        Keyword Args:
246            sort_by_damage: If enchants should be sorted by how much damage they add

249        async def _pred(card):
250            if await card.type_name() != "Enchantment":

254                if await effect.effect_type() == SpellEffects.modify_card_damage:
255                    return True

265            effect = (await card.get_spell_effects())[0]
266            return await effect.effect_param()

354        self,
355        name: str,
356        *,
357        on_member: str = None,
358        on_card: str = None,
359        on_client: bool = False,
360    ):
361        """
362        Attempt to cast a card
363
364        Args:
365            name: Name of the card to cast
366            on_member: Name of the member to cast the card on
367            on_card: Name of the card to cast the card on
368            on_client: Bool if the card should be cast on the client
369        """
370        try:
371            card = await self.get_card_named(name)
372
373            if on_member:
374                target = await self.get_member_named(on_member)
375                await card.cast(target)
376
377            elif on_card:
378                target = await self.get_card_named(on_card)
379                await card.cast(target)
380
381            elif on_client:
382                target = await self.get_client_member()
383                await card.cast(target)
384
385            else:
386                await card.cast(None)
387
388            return True
389
390        except ValueError:
391            return False

455    async def handle_round(self):
456        enchanted_aoes = await self.get_damaging_aoes(check_enchanted=True)
457        if enchanted_aoes:
458            await enchanted_aoes[0].cast(None)
459
460        unenchanted_aoes = await self.get_damaging_aoes(check_enchanted=False)
461        enchants = await self.get_damage_enchants(sort_by_damage=True)
462
463        # enchant card then cast card
464        if enchants and unenchanted_aoes:
465            await enchants[0].cast(unenchanted_aoes[0])
466            enchanted_aoes = await self.get_damaging_aoes(check_enchanted=True)
467
468            if not enchanted_aoes:
469                raise Exception("Enchant failure")
470
471            to_cast = enchanted_aoes[0]
472
473            if await to_cast.is_castable():
474                await to_cast.cast(None)
475
476        # no enchants so just cast card
477        elif not enchants and unenchanted_aoes:
478            to_cast = unenchanted_aoes[0]
479
480            if await to_cast.is_castable():
481                await to_cast.cast(None)
482
483        # hand full of enchants or enchants + other cards
484        elif enchants and not unenchanted_aoes:
485            if len(await self.get_cards()) == 7:
486                await enchants[0].discard()
487
488            # TODO: draw tc?
489            else:
490                raise Exception("No hits in hand")
491
492        # no enchants or aoes in hand
493        else:
494            # TODO: maybe flee instead?
495            if len(await self.get_cards()) == 0:
496                raise Exception("Out of cards")
497
498            # TODO: add method for people to subclass for this?
499            raise Exception("Out of hits and enchants")
