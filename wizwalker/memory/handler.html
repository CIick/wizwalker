 52    async def _get_open_autobot_address(self, size: int) -> int:
 53        if self._autobot_pos + size > self.AUTOBOT_SIZE:
 54            raise RuntimeError("Somehow went over autobot size")
 55
 56        addr = self._autobot_address + self._autobot_pos
 57        self._autobot_pos += size
 58
 59        logger.debug(
 60            f"Allocating autobot address {addr}; autobot position is now {self._autobot_pos}"
 61        )
 62        return addr

 64    async def _get_autobot_address(self):
 65        addr = await self.pattern_scan(
 66            self.AUTOBOT_PATTERN, module="WizardGraphicalClient.exe"
 67        )
 68        if addr is None:
 69            raise RuntimeError("Pattern scan failed for autobot pattern")
 70
 71        self._autobot_address = addr

 73    # noinspection PyTypeChecker
 74    async def _prepare_autobot(self):
 75        if self._autobot_address is None:
 76            await self._get_autobot_address()
 77
 78            # we only need to write back the pattern
 79            self._original_autobot_bytes = await self.read_bytes(
 80                self._autobot_address, len(self.AUTOBOT_PATTERN)
 81            )
 82            logger.debug(
 83                f"Got original bytes {self._original_autobot_bytes} from autobot"
 84            )
 85            await self.write_bytes(self._autobot_address, b"\x00" * self.AUTOBOT_SIZE)
 86
 87    async def _rewrite_autobot(self):
 88        if self._autobot_address is not None:
 89            compare_bytes = await self.read_bytes(
 90                self._autobot_address, len(self.AUTOBOT_PATTERN)
 91            )
 92            # Give some time for execution point to leave hooks
 93            await asyncio.sleep(0.5)
 94
 95            # Only write if the pattern isn't there
 96            if compare_bytes != self._original_autobot_bytes:
 97                logger.debug(
 98                    f"Rewriting bytes {self._original_autobot_bytes} to autobot"
 99                )
100                await self.write_bytes(
101                    self._autobot_address, self._original_autobot_bytes
102                )

104    async def _allocate_autobot_bytes(self, size: int) -> int:
105        address = await self._get_open_autobot_address(size)
106
107        return address

109    async def close(self):
110        for hook in self._active_hooks:
111            await hook.unhook()

120    async def _check_for_autobot(self):
121        if self._autobot_lock is None:
122            self._autobot_lock = asyncio.Lock()
123
124        # this is so it isn't prepared more than once at the same time
125        async with self._autobot_lock:
126            await self._prepare_autobot()

128    def _check_if_hook_active(self, hook_type) -> bool:
129        for hook in self._active_hooks:
130            if isinstance(hook, hook_type):
131                return True
132
133        return False

135    def _get_hook_by_type(self, hook_type) -> Any:
136        for hook in self._active_hooks:
137            if isinstance(hook, hook_type):
138                return hook
139
140        return None

142    async def _read_hook_base_addr(self, addr_name: str, hook_name: str):
143        addr = self._base_addrs.get(addr_name)
144        if addr is None:
145            raise HookNotActive(hook_name)
146
147        try:
148            return await self.read_typed(addr, "long long")
149        except pymem.exception.MemoryReadError:
150            raise HookNotReady(hook_name)

152    # wait for an addr to be set and not 0
153    async def _wait_for_value(self, address: int, timeout: int = None):
154        async def _wait_for_value_task():
155            while True:
156                try:
157                    value = await self.read_typed(address, "long long")
158                    logger.debug(
159                        f"Waiting for address {hex(address)}; got value {value}"
160                    )
161                except pymem.exception.MemoryReadError:
162                    pass
163                else:
164                    if value != 0:
165                        logger.debug(f"Address {hex(address)} is set")
166                        break
167                    else:
168                        logger.debug(f"Address {hex(address)} is not set yet; sleeping")
169                        await asyncio.sleep(0.5)
170
171        try:
172            await asyncio.wait_for(_wait_for_value_task(), timeout)
173        except TimeoutError:
174            # TODO: replace error
175            raise TimeoutError("Hook value took too long")

178    async def activate_all_hooks(
179        self, *, wait_for_ready: bool = True, timeout: float = None
180    ):
181        """
182        Activate all hooks but mouseless
183
184        Keyword Args:
185            wait_for_ready: Wait for hook values to be written
186            timeout: How long to wait for hook values to be written (None for no timeout)
187        """
188        await self.activate_player_hook(wait_for_ready=False)
189        # duel is only written to on battle join
190        await self.activate_duel_hook()
191        # quest hook is not written if the quest arrow is off
192        await self.activate_quest_hook()
193        await self.activate_player_stat_hook(wait_for_ready=False)
194        await self.activate_client_hook(wait_for_ready=False)
195        await self.activate_root_window_hook(wait_for_ready=False)
196        await self.activate_render_context_hook(wait_for_ready=False)
197        await self.activate_movement_teleport_hook(wait_for_ready=False)
198
199        if wait_for_ready:
200            wait_tasks = []
201            for atter_name in [
202                "player_struct",
203                "player_stat_struct",
204                "current_client",
205                "current_root_window",
206                "current_render_context",
207            ]:
208                value = self._base_addrs[atter_name]
209                wait_tasks.append(
210                    asyncio.create_task(self._wait_for_value(value, timeout))
211                )
212
213            await asyncio.gather(*wait_tasks)

 55            raise RuntimeError("Somehow went over autobot size")
 56
 57        addr = self._autobot_address + self._autobot_pos
 58        self._autobot_pos += size
 59
 60        logger.debug(
 61            f"Allocating autobot address {addr}; autobot position is now {self._autobot_pos}"
 62        )
 63        return addr

 65    async def _get_autobot_address(self):
 66        addr = await self.pattern_scan(
 67            self.AUTOBOT_PATTERN, module="WizardGraphicalClient.exe"
 68        )
 69        if addr is None:
 70            raise RuntimeError("Pattern scan failed for autobot pattern")
 71
 72        self._autobot_address = addr

 74    # noinspection PyTypeChecker
 75    async def _prepare_autobot(self):
 76        if self._autobot_address is None:
 77            await self._get_autobot_address()
 78
 79            # we only need to write back the pattern
 80            self._original_autobot_bytes = await self.read_bytes(
 81                self._autobot_address, len(self.AUTOBOT_PATTERN)
 82            )
 83            logger.debug(
 84                f"Got original bytes {self._original_autobot_bytes} from autobot"
 85            )
 86            await self.write_bytes(self._autobot_address, b"\x00" * self.AUTOBOT_SIZE)
 87
 88    async def _rewrite_autobot(self):
 89        if self._autobot_address is not None:
 90            compare_bytes = await self.read_bytes(
 91                self._autobot_address, len(self.AUTOBOT_PATTERN)
 92            )
 93            # Give some time for execution point to leave hooks
 94            await asyncio.sleep(0.5)
 95
 96            # Only write if the pattern isn't there
 97            if compare_bytes != self._original_autobot_bytes:
 98                logger.debug(
 99                    f"Rewriting bytes {self._original_autobot_bytes} to autobot"
100                )
101                await self.write_bytes(
102                    self._autobot_address, self._original_autobot_bytes
103                )

106        address = await self._get_open_autobot_address(size)
107
108        return address

154    async def _wait_for_value(self, address: int, timeout: int = None):
155        async def _wait_for_value_task():
156            while True:
157                try:
158                    value = await self.read_typed(address, "long long")
159                    logger.debug(
160                        f"Waiting for address {hex(address)}; got value {value}"
161                    )
162                except pymem.exception.MemoryReadError:
163                    pass
164                else:
165                    if value != 0:
166                        logger.debug(f"Address {hex(address)} is set")
167                        break
168                    else:
169                        logger.debug(f"Address {hex(address)} is not set yet; sleeping")
170                        await asyncio.sleep(0.5)
171
172        try:
173            await asyncio.wait_for(_wait_for_value_task(), timeout)
174        except TimeoutError:
175            # TODO: replace error
176            raise TimeoutError("Hook value took too long")
