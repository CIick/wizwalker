161    async def get_jump_bytecode(self) -> bytes:
162        distance = self.hook_address - self.jump_address
163
164        relitive_jump = distance - 5
165        packed_relitive_jump = struct.pack("<i", relitive_jump)
166
167        return b"\xE9" + packed_relitive_jump + (b"\x90" * self.noops)

169    async def bytecode_generator(self, packed_exports):

172    async def get_hook_bytecode(self) -> bytes:
173        packed_exports = []
174        for export in self.exports:
175            # addr = self.alloc(export[1])
176            addr = self.hook_handler.process.allocate(export[1])
177            setattr(self, export[0], addr)
178            packed_addr = struct.pack("<Q", addr)
179            packed_exports.append((export[0], packed_addr))
180
181        bytecode = await self.bytecode_generator(packed_exports)
182
183        return_addr = self.jump_address + self.instruction_length
184
185        relitive_return_jump = return_addr - (self.hook_address + len(bytecode)) - 5
186        packed_relitive_return_jump = struct.pack("<i", relitive_return_jump)
187
188        bytecode += b"\xE9" + packed_relitive_return_jump
189
190        return bytecode
191
192    async def unhook(self):
193        await super().unhook()
194        for export in self.exports:
195            if getattr(self, export[0], None):
196                await self.free(getattr(self, export[0]))
