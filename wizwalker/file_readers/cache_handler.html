 36    async def _check_updated(
 37        self, wad_file: Wad, files: Union[List[str], str]
 38    ) -> List[str]:
 39        if isinstance(files, str):
 40            files = [files]
 41
 42        if not self._wad_cache:
 43            self._wad_cache = await self.get_wad_cache()
 44
 45        res = []
 46
 47        for file_name in files:
 48            file_info = await wad_file.get_file_info(file_name)
 49
 50            if self._wad_cache[wad_file.name][file_name] != file_info.size:
 51                logger.info(
 52                    f"{file_name} has updated. old: {self._wad_cache[wad_file.name][file_name]} new: {file_info.size}"
 53                )
 54                res.append(file_name)
 55                self._wad_cache[wad_file.name][file_name] = file_info.size
 56
 57            else:
 58                logger.info(f"{file_name} has not updated from {file_info.size}")
 59
 60        return res

113    @staticmethod
114    def _parse_lang_file(file_data):
115        try:
116            decoded = file_data.decode("utf-16")
117
118        # empty file
119        except UnicodeDecodeError:
120            return
121
122        # splitlines splits whitespace that lang files should not recognize as a newline
123        file_lines = decoded.split("\r\n")
124
125        header, *lines = file_lines
126        _, lang_name = header.split(":")
127
128        lang_mapping = dict(zip(lines[::3], lines[2::3]))
129
130        return {lang_name: lang_mapping}

249    async def get_langcode_name(self, langcode: str):
250        """
251        Get the langcode name from the langcode i.e Spells_00001
252
253        Args:
254            langcode: Langcode in the format filename_code
255
256        Raises:
257            ValueError: If the langcode does not have a match
258
259        """

266        cached = False

273        if not cached:

275
276        langcode_map = await self.get_langcode_map()
277        lang_file = langcode_map.get(lang_filename)
278
279        if lang_file is None:
280            raise ValueError(f"No lang file named {lang_filename}")
281
282        lang_name = lang_file.get(code)
283
284        if lang_name is None:
285            raise ValueError(f"No lang name with code {code}")
286
